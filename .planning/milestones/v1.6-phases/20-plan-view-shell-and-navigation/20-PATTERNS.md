**Inline edit pattern — no codebase analog (new pattern):**

No existing inline edit component exists in this codebase. Implement as a controlled `<input>` using the cancellingRef pattern from RESEARCH.md Pattern 3. The critical requirement is guarding `onBlur` against Escape via a `useRef` flag:

```tsx
// RESEARCH.md Pattern 3 — use this directly; no codebase analog to reference
const cancellingRef = useRef(false);

function handleKeyDown(e: React.KeyboardEvent<HTMLInputElement>) {
  if (e.key === "Enter") {
    const trimmed = e.currentTarget.value.trim();
    if (trimmed) onCommitRename(trimmed);  // non-empty: commit
    else onCancelRename();                 // D-09: empty = cancel
  }
  if (e.key === "Escape") {
    cancellingRef.current = true;          // flag before blur fires
    onCancelRename();
    e.currentTarget.blur();
  }
}

function handleBlur(e: React.FocusEvent<HTMLInputElement>) {
  if (cancellingRef.current) {
    cancellingRef.current = false;
    return;                                // Escape already handled
  }
  const trimmed = e.currentTarget.value.trim();
  if (trimmed) onCommitRename(trimmed);
  else onCancelRename();                   // D-09: empty = cancel
}

// Select-all on mount (D-07/D-08: pre-fill + all text selected)
const inputRef = useRef<HTMLInputElement>(null);
useEffect(() => { inputRef.current?.select(); }, []);

<input
  ref={inputRef}
  type="text"
  autoFocus
  defaultValue={initialValue}
  aria-label="Plan name"
  className="text-sm bg-transparent border-b border-border focus:outline-none w-full"
  onKeyDown={handleKeyDown}
  onBlur={handleBlur}
/>
```
