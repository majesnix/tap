**Step 2 — Implement src/components/form/OneofField.tsx (replace stub body, keep interface):**

CRITICAL PATH CONVENTION: Branch fields are rendered at `${fieldPath}.${branchField.name}` — a flat
path directly under the oneof group object. This matches the form values shape:
  `{ payment: { _selected: "card_number", card_number: "" } }`
and the Rust encoder's oneof group handling. Do NOT add an extra `.${name}` segment in the middle.

```typescript
import React, { useEffect } from "react";
import { Controller, useWatch, useFormContext } from "react-hook-form";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import type { FieldSchema, RenderFieldFn } from "@/lib/types";

export interface OneofFieldProps {
  field: FieldSchema;
  fieldPath: string;
  depth: number;
  renderBranchField: RenderFieldFn;
}

export function OneofField({ field, fieldPath, depth, renderBranchField }: OneofFieldProps) {
  const { control, unregister } = useFormContext();
  const branches = (field.kind as { type: "oneof"; branches: FieldSchema[][] }).branches;

  // Branch name = first field's name in each branch (or oneof_group)
  const branchNames = branches.map((branch) => branch[0]?.name ?? "unknown");
  const firstBranch = branchNames[0] ?? "";

  const selected = useWatch({
    control,
    name: `${fieldPath}._selected`,
    defaultValue: firstBranch,
  });

  // When selected branch changes: unregister all non-selected branches (proto wire semantics)
  useEffect(() => {
    branchNames.forEach((name) => {
      if (name !== selected) {
        unregister(`${fieldPath}.${name}`);
      }
    });
  }, [selected, fieldPath, unregister, branchNames]);

  return (
    <div className="flex flex-col gap-2 mb-3">
      <div className="flex items-center gap-2">
        <Label className="text-xs font-semibold">{field.label}</Label>
        <Badge variant="outline" className="text-xs px-1.5 py-0">oneof</Badge>
      </div>
      <Controller
        name={`${fieldPath}._selected`}
        control={control}
        defaultValue={firstBranch}
        render={({ field: rhfField }) => (
          <RadioGroup
            value={rhfField.value}
            onValueChange={rhfField.onChange}
            className="flex flex-col gap-1"
          >
            {branchNames.map((name) => (
              <div key={name} className="flex items-center gap-2">
                <RadioGroupItem value={name} id={`${fieldPath}._${name}`} />
                <Label htmlFor={`${fieldPath}._${name}`} className="text-sm">
                  {name}
                </Label>
              </div>
            ))}
          </RadioGroup>
        )}
      />
      {/* Only render the selected branch — conditional mount, not CSS hide.
          Path convention: ${fieldPath}.${branchField.name} — flat, not double-nested.
          This matches the Rust encoder's expected { payment: { card_number: "" } } shape. */}
      {branchNames.map((name, idx) =>
        selected === name ? (
          <div key={name} className="ml-4 border-l border-border pl-3">
            {branches[idx]?.map((branchField) =>
              renderBranchField(branchField, `${fieldPath}.${branchField.name}`, depth)
            )}
          </div>
        ) : null
      )}
    </div>
  );
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | OneofField — _selected value could be forged to a non-existent branch name | mitigate | Rust `encode_message` receives the oneof form values JSON. The encoder looks up the selected branch field by name in `MessageDescriptor.get_field_by_name()` — if `_selected` names a field that doesn't exist in the descriptor, `get_field_by_name` returns `None` and the oneof group is silently skipped (no encoding, no data corruption). Worst case: no oneof field is set in the wire format, which is valid protobuf. |
| T-04-02 | Tampering | EnumField — JavaScript sends a number value; Rust validates it against enum descriptor | mitigate | Rust `encode_message` validates enum numbers against the descriptor's `EnumDescriptor.get_value_by_number()`. An invalid enum number (not in the proto definition) causes `AppError::EncodeError { field, message }` which the frontend surfaces as an inline error. |
</threat_model>
