```rust
pub fn extract_schema(pool: &prost_reflect::DescriptorPool) -> crate::schema::types::ProtoSchema {
    use std::collections::HashMap;
    let messages: Vec<_> = pool
        .all_messages()
        .filter(|m| m.parent_message().is_none())
        .map(|m| extract_message_schema(&m))
        .collect();
    let message_map: HashMap<String, _> = messages.iter()
        .map(|m: &crate::schema::types::MessageSchema| (m.full_name.clone(), m.clone()))
        .collect();
    crate::schema::types::ProtoSchema { messages, message_map }
}
```

`extract_message_schema(msg: &MessageDescriptor)` groups fields by oneof:
1. Collect all fields from `msg.fields()`
2. Separate oneof fields (those with `field.containing_oneof().is_some()`) from regular fields
3. Group oneof fields by `field.containing_oneof().unwrap().name().to_string()`
4. For each oneof group, produce one `FieldSchema` with `FieldKind::Oneof { branches }` where `branches` = `vec![vec![field_schema], ...]` (one Vec per branch, each branch has one FieldSchema in a simple oneof)
5. Non-oneof fields map directly via `extract_field_schema`

`extract_field_schema(field: &FieldDescriptor)` maps Kind:
- WKT detection FIRST (before general message check): if `field.kind()` is `Kind::Message(msg)` and `msg.full_name() == "google.protobuf.Timestamp"` → `FieldKind::WellKnown { wkt: "Timestamp".to_string() }`; `"google.protobuf.Duration"` → `WellKnown { wkt: "Duration".to_string() }`
- `Kind::Bool` → `Scalar { scalar: ScalarKind::Bool }`
- `Kind::String` → `Scalar { scalar: ScalarKind::String }`
- `Kind::Bytes` → `Scalar { scalar: ScalarKind::Bytes }`
- `Kind::Int32` → `ScalarKind::Int32`, `Kind::Sint32` → `Sint32`, `Kind::Sfixed32` → `Sfixed32`, `Kind::Fixed32` → `Fixed32`
- `Kind::Int64` → `Int64`, `Kind::Sint64` → `Sint64`, `Kind::Sfixed64` → `Sfixed64`, `Kind::Fixed64` → `Fixed64`
- `Kind::Uint32` → `Uint32`, `Kind::Uint64` → `Uint64`
- `Kind::Float` → `Float`, `Kind::Double` → `Double`
- `Kind::Enum(enum_desc)` → `Enum { values: enum_desc.values().map(|v| EnumValue { name: v.name().to_string(), number: v.number() }).collect() }`
- `Kind::Message(msg_desc)` (non-WKT) → `Message { full_name: msg_desc.full_name().to_string() }`
- `repeated`: `field.cardinality() == Cardinality::Repeated`
- `oneof_group`: `field.containing_oneof().map(|o| o.name().to_string())`
- `default_value`: convert `field.default_value()` (a `prost_reflect::Value`) to `serde_json::Value` using a match: `Value::Bool(b)` → `json!(b)`, `Value::I32(n)` → `json!(n)`, `Value::I64(n)` → `json!(n)`, `Value::U32(n)` → `json!(n)`, `Value::U64(n)` → `json!(n)`, `Value::F32(f)` → `json!(f)`, `Value::F64(f)` → `json!(f)`, `Value::String(s)` → `json!(s)`, `Value::Bytes(_)` → `json!("")`, `Value::EnumNumber(n)` → `json!(n)`, `Value::Message(_)` → `json!(null)`, `Value::List(_)` → `json!([])`, `Value::Map(_)` → `json!(null)`.

```rust
use crate::schema::{extractor, types::ProtoSchema};
use crate::error::AppError;
use std::sync::Mutex;

#[tauri::command]
pub async fn parse_proto(
    file_path: String,
    include_paths: Vec<String>,
    pool_state: tauri::State<'_, Mutex<Option<prost_reflect::DescriptorPool>>>,
) -> Result<ProtoSchema, AppError> {
    let mut compiler = protox::Compiler::new(&include_paths)
        .map_err(|e| AppError::ParseError(e.to_string()))?;
    compiler.include_imports(true);
    compiler.open_file(&file_path)
        .map_err(|e| AppError::ParseError(e.to_string()))?;
    let fds = compiler.file_descriptor_set();
    let pool = prost_reflect::DescriptorPool::from_file_descriptor_set(fds)
        .map_err(|e| AppError::ParseError(e.to_string()))?;
    // Store pool in State for encode_message to reuse (DescriptorPool is Arc-backed, clone is O(1))
    *pool_state.lock().unwrap() = Some(pool.clone());
    let schema = extractor::extract_schema(&pool);
    Ok(schema)
}
```

**Step 10 — Write src-tauri/src/commands/encode.rs:**

The encoder handles all field kinds including nested messages, oneof groups, and WellKnownTypes. The `encode_message` function has two passes over the JSON object: the standard field pass (using `get_field_by_name`) and an oneof fallback pass (for keys that are oneof group names, not individual field names).

```rust
use crate::error::AppError;
use std::sync::Mutex;

#[tauri::command]
pub async fn encode_message(
    message_type: String,
    form_values: serde_json::Value,
    pool_state: tauri::State<'_, Mutex<Option<prost_reflect::DescriptorPool>>>,
) -> Result<Vec<u8>, AppError> {
    let pool_guard = pool_state.lock().unwrap();
    let pool = pool_guard.as_ref().ok_or_else(|| {
        AppError::ParseError("No proto file loaded. Call parse_proto first.".to_string())
    })?;

    let msg_desc = pool.get_message_by_name(&message_type).ok_or_else(|| {
        AppError::ParseError(format!("Message type '{}' not found in descriptor pool", message_type))
    })?;

    let mut dynamic_msg = prost_reflect::DynamicMessage::new(msg_desc.clone());
    populate_message(&mut dynamic_msg, &msg_desc, &form_values)
        .map_err(|e| AppError::EncodeError { field: message_type.clone(), message: e })?;

    use prost::Message;
    let mut buf = Vec::new();
    dynamic_msg.encode(&mut buf)
        .map_err(|e| AppError::ParseError(format!("Encode failed: {}", e)))?;
    Ok(buf)
}

/// Populate a DynamicMessage from a JSON object, handling regular fields and oneof groups.
fn populate_message(
    msg: &mut prost_reflect::DynamicMessage,
    msg_desc: &prost_reflect::MessageDescriptor,
    val: &serde_json::Value,
) -> Result<(), String> {
    let obj = match val.as_object() {
        Some(o) => o,
        None => return Ok(()), // null or non-object — leave message at defaults
    };

    for (key, field_val) in obj {
        if key == "_selected" {
            continue; // oneof discriminant key — handled below
        }

        if let Some(field_desc) = msg_desc.get_field_by_name(key) {
            // Regular field or individual oneof branch field
            set_field_value(msg, &field_desc, field_val)
                .map_err(|e| format!("{}.{}: {}", msg_desc.full_name(), key, e))?;
        } else {
            // Not a direct field name — check if it is a oneof group name.
            // Form shape for oneof:  { payment: { _selected: "card_number", card_number: "" } }
            // prost-reflect has no get_field_by_name("payment") — payment is not a field name.
            if let Some(oneof_desc) = msg_desc.oneofs().find(|o| o.name() == key.as_str()) {
                if let serde_json::Value::Object(oneof_obj) = field_val {
                    if let Some(serde_json::Value::String(selected)) = oneof_obj.get("_selected") {
                        if let Some(branch_field) = msg_desc.get_field_by_name(selected.as_str()) {
                            // Verify the branch field belongs to this oneof
                            if branch_field.containing_oneof().map(|o| o.name() == oneof_desc.name()).unwrap_or(false) {
                                if let Some(branch_val) = oneof_obj.get(selected.as_str()) {
                                    set_field_value(msg, &branch_field, branch_val)
                                        .map_err(|e| format!("{}.{}.{}: {}", msg_desc.full_name(), key, selected, e))?;
                                }
                            }
                        }
                    }
                }
            }
            // If neither a field nor a oneof group, silently skip (unknown key from form)
        }
    }
    Ok(())
}

fn set_field_value(
    msg: &mut prost_reflect::DynamicMessage,
    field: &prost_reflect::FieldDescriptor,
    val: &serde_json::Value,
) -> Result<(), String> {
    use prost_reflect::{Value, Kind, Cardinality};

    // Handle repeated fields: val must be a JSON array
    if field.cardinality() == Cardinality::Repeated {
        let arr = val.as_array().map(|a| a.as_slice()).unwrap_or(&[]);
        let items: Vec<Value> = arr.iter()
            .map(|item| scalar_or_message_value(field, item))
            .collect::<Result<_, _>>()?;
        msg.set_field_by_name(field.name(), Value::List(items));
        return Ok(());
    }

    let prost_val = scalar_or_message_value(field, val)?;
    msg.set_field_by_name(field.name(), prost_val);
    Ok(())
}

/// Convert a single JSON value to a prost_reflect::Value for the given field kind.
/// Handles scalars, enums, nested messages (recursive), and WellKnownTypes.
fn scalar_or_message_value(
    field: &prost_reflect::FieldDescriptor,
    val: &serde_json::Value,
) -> Result<prost_reflect::Value, String> {
    use prost_reflect::{Value, Kind};

    Ok(match field.kind() {
        Kind::Bool => Value::Bool(val.as_bool().unwrap_or(false)),
        Kind::Int32 | Kind::Sint32 | Kind::Sfixed32 =>
            Value::I32(val.as_i64().unwrap_or(0) as i32),
        Kind::Int64 | Kind::Sint64 | Kind::Sfixed64 => {
            let n: i64 = val.as_str()
                .and_then(|s| s.parse::<i64>().ok())
                .or_else(|| val.as_i64())
                .unwrap_or(0);
            Value::I64(n)
        }
        Kind::Uint32 | Kind::Fixed32 =>
            Value::U32(val.as_u64().unwrap_or(0) as u32),
        Kind::Uint64 | Kind::Fixed64 => {
            let n: u64 = val.as_str()
                .and_then(|s| s.parse::<u64>().ok())
                .or_else(|| val.as_u64())
                .unwrap_or(0);
            Value::U64(n)
        }
        Kind::Float => Value::F32(val.as_f64().unwrap_or(0.0) as f32),
        Kind::Double => Value::F64(val.as_f64().unwrap_or(0.0)),
        Kind::String => Value::String(val.as_str().unwrap_or("").to_string()),
        Kind::Bytes => {
            let s = val.as_str().unwrap_or("");
            Value::Bytes(base64_decode_or_empty(s).into())
        }
        Kind::Enum(_) =>
            Value::EnumNumber(val.as_i64().unwrap_or(0) as i32),
        Kind::Message(nested_desc) => {
            // WellKnownType Timestamp: form sends an ISO-8601 datetime-local string
            // ("2024-01-15T10:30") → encode as nested {seconds, nanos} message
            if nested_desc.full_name() == "google.protobuf.Timestamp" {
                let mut ts_msg = prost_reflect::DynamicMessage::new(nested_desc.clone());
                let (seconds, nanos) = parse_datetime_to_epoch(val.as_str().unwrap_or(""));
                if let Some(f) = nested_desc.get_field_by_name("seconds") {
                    ts_msg.set_field_by_name(f.name(), Value::I64(seconds));
                }
                if let Some(f) = nested_desc.get_field_by_name("nanos") {
                    ts_msg.set_field_by_name(f.name(), Value::I32(nanos));
                }
                return Ok(Value::Message(ts_msg));
            }

            // WellKnownType Duration: form sends a human string like "1h30m", "90s", or raw seconds
            if nested_desc.full_name() == "google.protobuf.Duration" {
                let mut dur_msg = prost_reflect::DynamicMessage::new(nested_desc.clone());
                let (seconds, nanos) = parse_duration_string(val.as_str().unwrap_or(""));
                if let Some(f) = nested_desc.get_field_by_name("seconds") {
                    dur_msg.set_field_by_name(f.name(), Value::I64(seconds));
                }
                if let Some(f) = nested_desc.get_field_by_name("nanos") {
                    dur_msg.set_field_by_name(f.name(), Value::I32(nanos));
                }
                return Ok(Value::Message(dur_msg));
            }

            // General nested message: val is a JSON object — recurse
            let mut nested_msg = prost_reflect::DynamicMessage::new(nested_desc.clone());
            populate_message(&mut nested_msg, &nested_desc, val)
                .map_err(|e| format!("nested {}: {}", nested_desc.full_name(), e))?;
            Value::Message(nested_msg)
        }
    })
}

/// Parse an ISO-8601 datetime-local string ("2024-01-15T10:30" or "2024-01-15T10:30:00")
/// into Unix epoch (seconds, nanos). Returns (0, 0) on parse failure.
fn parse_datetime_to_epoch(s: &str) -> (i64, i32) {
    // Minimal parser: "YYYY-MM-DDTHH:MM" or "YYYY-MM-DDTHH:MM:SS"
    // Accepts the output of <input type="datetime-local">
    let parts: Vec<&str> = s.splitn(2, 'T').collect();
    if parts.len() != 2 { return (0, 0); }
    let date_parts: Vec<&str> = parts[0].split('-').collect();
    let time_parts: Vec<&str> = parts[1].splitn(3, ':').collect();
    if date_parts.len() < 3 || time_parts.len() < 2 { return (0, 0); }

    let year: i64 = date_parts[0].parse().unwrap_or(1970);
    let month: i64 = date_parts[1].parse().unwrap_or(1);
    let day: i64 = date_parts[2].parse().unwrap_or(1);
    let hour: i64 = time_parts[0].parse().unwrap_or(0);
    let minute: i64 = time_parts[1].parse().unwrap_or(0);
    let second: i64 = time_parts.get(2)
        .and_then(|s| s.parse().ok())
        .unwrap_or(0);

    // Simplified Julian Day → Unix timestamp calculation (Gregorian calendar)
    // Uses the same formula as the C standard library mktime for UTC
    let ym = (month - 14) / 12;
    let jdn = (1461 * (year + 4800 + ym)) / 4
        + (367 * (month - 2 - 12 * ym)) / 12
        - (3 * ((year + 4900 + ym) / 100)) / 4
        + day
        - 32075;
    // Unix epoch starts at JDN 2440588 (1970-01-01)
    let unix_days = jdn - 2440588;
    let seconds = unix_days * 86400 + hour * 3600 + minute * 60 + second;
    (seconds, 0)
}

/// Parse a human-readable duration string ("1h30m", "90s", "5m30s", "3600") into (seconds, nanos).
/// Accepts: h (hours), m (minutes), s (seconds). Raw integer is treated as seconds.
/// Returns (0, 0) on empty or unparseable input.
fn parse_duration_string(s: &str) -> (i64, i32) {
    if s.is_empty() { return (0, 0); }
    // Try raw seconds first (plain integer)
    if let Ok(n) = s.trim().parse::<i64>() {
        return (n, 0);
    }
    let mut total_seconds: i64 = 0;
    let mut current = String::new();
    for ch in s.chars() {
        if ch.is_ascii_digit() {
            current.push(ch);
        } else {
            let n: i64 = current.parse().unwrap_or(0);
            current.clear();
            match ch {
                'h' => total_seconds += n * 3600,
                'm' => total_seconds += n * 60,
                's' => total_seconds += n,
                _ => {} // ignore unknown suffix
            }
        }
    }
    // Trailing digits without unit — treat as seconds
    if !current.is_empty() {
        total_seconds += current.parse::<i64>().unwrap_or(0);
    }
    (total_seconds, 0)
}

fn base64_decode_or_empty(s: &str) -> Vec<u8> {
    let cleaned = s.replace(['\n', '\r', ' '], "");
    let mut result = Vec::new();
    let alphabet = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
    let mut buffer = 0u32;
    let mut bits = 0u32;
    for c in cleaned.bytes() {
        if c == b'=' { break; }
        let pos = match alphabet.iter().position(|&x| x == c) {
            Some(p) => p as u32,
            None => return Vec::new(),
        };
        buffer = (buffer << 6) | pos;
        bits += 6;
        if bits >= 8 {
            bits -= 8;
            result.push((buffer >> bits) as u8);
        }
    }
    result
}
```

**Step 11 — Write src/components/form/ProtoFormRenderer.tsx — FINAL DISPATCH (never modified by Wave 2):**
```tsx
import React from "react";
import { FormProvider, useForm } from "react-hook-form";
import type { MessageSchema, FieldSchema, ProtoSchema } from "@/lib/types";
import { ScalarField } from "./ScalarField";
import { NestedMessageField } from "./NestedMessageField";
import { RepeatedField } from "./RepeatedField";
import { EnumField } from "./EnumField";
import { OneofField } from "./OneofField";
import { WellKnownTypeField } from "./WellKnownTypeField";
import { DepthCapPlaceholder } from "./DepthCapPlaceholder";

interface ProtoFormRendererProps {
  messageSchema: MessageSchema;
  fieldPath: string;
  depth: number;
  schema: ProtoSchema;
}

function buildDefaultValues(messageSchema: MessageSchema): Record<string, unknown> {
  const defaults: Record<string, unknown> = {};
  for (const field of messageSchema.fields) {
    if (field.default_value !== undefined) {
      defaults[field.name] = field.default_value;
    } else if (field.kind.type === "scalar") {
      const scalar = field.kind.scalar;
      if (scalar === "bool") defaults[field.name] = false;
      else if (["int64", "uint64", "sint64", "sfixed64", "fixed64"].includes(scalar)) defaults[field.name] = "0";
      else if (["string", "bytes"].includes(scalar)) defaults[field.name] = "";
      else defaults[field.name] = 0;
    } else if (field.repeated) {
      defaults[field.name] = [];
    } else {
      defaults[field.name] = null;
    }
  }
  return defaults;
}

export function ProtoFormRenderer({ messageSchema, fieldPath, depth, schema }: ProtoFormRendererProps) {
  const methods = useForm({ defaultValues: buildDefaultValues(messageSchema) });

  // renderField is the recursive dispatch function — passed as prop to child components that recurse
  // This breaks circular imports: child components NEVER import ProtoFormRenderer
  function renderField(field: FieldSchema, path: string, currentDepth: number): React.ReactNode {
    if (currentDepth >= 5) return <DepthCapPlaceholder key={path} />;

    if (field.repeated) {
      return (
        <RepeatedField
          key={path}
          field={field}
          fieldPath={path}
          depth={currentDepth}
          schema={schema}
          renderItem={(f, p, d) => renderField(f, p, d)}
        />
      );
    }

    switch (field.kind.type) {
      case "scalar":
        return <ScalarField key={path} field={field} fieldPath={path} />;
      case "message":
        return (
          <NestedMessageField
            key={path}
            field={field}
            fieldPath={path}
            depth={currentDepth}
            schema={schema}
            renderChildField={(f, p, d) => renderField(f, p, d)}
          />
        );
      case "enum":
        return <EnumField key={path} field={field} fieldPath={path} />;
      case "oneof":
        return (
          <OneofField
            key={path}
            field={field}
            fieldPath={path}
            depth={currentDepth}
            renderBranchField={(f, p, d) => renderField(f, p, d)}
          />
        );
      case "well_known":
        return <WellKnownTypeField key={path} field={field} fieldPath={path} />;
      default:
        return null;
    }
  }

  return (
    <FormProvider {...methods}>
      <form>
        {messageSchema.fields.map((field) =>
          renderField(field, fieldPath ? `${fieldPath}.${field.name}` : field.name, depth)
        )}
      </form>
    </FormProvider>
  );
}
```
