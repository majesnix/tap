```rust
/// Build a percent-encoded AMQP URI.
/// SECURITY: The returned String contains the cleartext password — use immediately,
/// do NOT store in any field, log file, or tracing output. Drop after use.
pub fn build_amqp_uri(host: &str, port: u16, vhost: &str, user: &str, pass: &str) -> String {
    let enc_vhost = utf8_percent_encode(vhost, NON_ALPHANUMERIC);
    let enc_pass  = utf8_percent_encode(pass,  NON_ALPHANUMERIC);
    // "/" vhost → "%2F"; "@" in password → "%40"
    format!("amqp://{}:{}@{}:{}/{}", user, enc_pass, host, port, enc_vhost)
}

#[cfg(test)]
mod uri_tests {
    use super::*;

    #[test]
    fn default_vhost_encodes_correctly() {
        let uri = build_amqp_uri("localhost", 5672, "/", "guest", "guest");
        assert!(uri.contains("%2F"), "default vhost '/' must become '%2F' in URI");
        assert!(!uri.contains("//"), "unencoded '/' in path causes wrong vhost");
    }

    #[test]
    fn special_chars_in_password_encoded() {
        let uri = build_amqp_uri("localhost", 5672, "/", "user", "p@ss:w0rd#");
        assert!(!uri.contains("@p"), "bare '@' in password would break URI parsing");
    }
}
```

```rust
/// Test an AMQP connection for a saved profile.
/// Ephemeral: connects, opens a channel, closes. No persistent state kept.
/// SECURITY: AMQP URI is built inside this function, used, and dropped.
/// Never logged — tracing only records profile name and success/failure.
/// Wraps lapin connect in a 10s timeout to prevent indefinite hang on unreachable brokers.
#[tauri::command]
pub async fn test_connection(app: AppHandle, profile_name: String) -> Result<(), AppError> {
    // Load profile from store
    let store = app
        .store("tap.json")
        .map_err(|e| AppError::StoreError(e.to_string()))?;

    let profiles: Vec<crate::profiles::ConnectionProfile> = store
        .get(crate::profiles::PROFILES_STORE_KEY)
        .and_then(|v| serde_json::from_value(v).ok())
        .unwrap_or_default();

    let profile = profiles
        .into_iter()
        .find(|p| p.name == profile_name)
        .ok_or_else(|| AppError::ProfileNotFound(profile_name.clone()))?;

    // Load password from OS keychain (NOT from store JSON)
    let password = get_password(&profile_name)?;

    // Build AMQP URI — use immediately, drop after connection attempt
    let uri = build_amqp_uri(&profile.host, profile.port, &profile.vhost, &profile.username, &password);

    // Connect with 10s timeout to prevent indefinite hang
    // SECURITY: uri contains cleartext password — never log it
    let conn = tokio::time::timeout(
        Duration::from_secs(10),
        Connection::connect(&uri, ConnectionProperties::default()),
    )
    .await
    .map_err(|_| AppError::AmqpError("Connection timed out (10s)".to_string()))?
    .map_err(|e| AppError::AmqpError(e.to_string()))?;

    // Open a channel to verify credentials and vhost access
    conn.create_channel()
        .await
        .map_err(|e| AppError::AmqpError(e.to_string()))?;

    // Close the connection — ephemeral pattern, no persistent state
    let _ = conn.close(0, "").await;

    tracing::debug!("Connection test passed for profile: {}", profile_name);
    Ok(())
}

/// Activate a profile: test its connection and update frontend connection state.
/// Called when user switches profiles in the sidebar dropdown.
/// Returns Ok(()) on success — frontend updates status dot via store on Ok/Err.
#[tauri::command]
pub async fn activate_profile(app: AppHandle, profile_name: String) -> Result<(), AppError> {
    // Reuse test_connection logic — if it succeeds, this profile is now active
    test_connection(app, profile_name).await
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Rust → RabbitMQ TCP | AMQP credentials leave the process; URI must not be logged |
| Rust keyring → AMQP URI | Password retrieved from keychain, encoded, embedded in URI, used once, dropped |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-06 | Information Disclosure | AMQP URI in `test_connection` | mitigate | URI is a local `let uri` variable inside the async fn. Never passed to tracing, never stored in a struct or app state. `let _ = conn.close().await` — URI is dropped when the fn returns. |
| T-02-07 | Denial of Service | `Connection::connect` on unreachable broker | mitigate | Wrapped in `tokio::time::timeout(Duration::from_secs(10), ...)`. Times out with `AmqpError("Connection timed out (10s)")` — UI shows inline error, not a hang. |
| T-02-08 | Information Disclosure | `AppError::AmqpError(e.to_string())` — lapin error may include host:port | accept | The error string from lapin includes host:port but NOT credentials. Displaying the host in an error message is acceptable (user entered it). Password is never in lapin error strings. |
| T-02-09 | Spoofing | Stale profile after rename/delete | accept | Profile names are user-controlled unique keys. No auth token or session involved — spoofing is not applicable. |
</threat_model>
