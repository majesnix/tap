The `activeFilePath` and `schema` fields are plain stored state — NOT JavaScript getter accessors. They are always set explicitly inside `addOrActivateFile`, `closeFile`, `setActiveIndex`, and `reset`. Consumers can select them directly: `useProtoStore((s) => s.activeFilePath)`.

Example store shape:
```typescript
const INITIAL_STATE = {
  openFiles: [] as Array<{ filePath: string; schema: ProtoSchema }>,
  activeIndex: -1,
  activeFilePath: null as string | null,
  schema: null as ProtoSchema | null,
  selectedMessageType: null as string | null,
  hexPreview: "",
  isEncoding: false,
  encodeError: null as string | null,
  latestValues: null as Record<string, unknown> | null,
  lastSendAt: null as number | null,
  pendingReplayValues: null as Record<string, unknown> | null,
};

export const useProtoStore = create<ProtoStore>((set) => ({
  ...INITIAL_STATE,
  addOrActivateFile: (filePath, schema) =>
    set((s) => {
      const existing = s.openFiles.findIndex((f) => f.filePath === filePath);
      if (existing !== -1) {
        return {
          activeIndex: existing,
          activeFilePath: s.openFiles[existing].filePath,
          schema: s.openFiles[existing].schema,
          selectedMessageType: s.openFiles[existing].schema.messages[0]?.full_name ?? null,
          hexPreview: "",
          encodeError: null,
        };
      }
      const newFiles = [...s.openFiles, { filePath, schema }];
      const newIndex = newFiles.length - 1;
      return {
        openFiles: newFiles,
        activeIndex: newIndex,
        activeFilePath: filePath,
        schema,
        selectedMessageType: schema.messages[0]?.full_name ?? null,
        hexPreview: "",
        encodeError: null,
      };
    }),
  closeFile: (index) =>
    set((s) => {
      const newFiles = s.openFiles.filter((_, i) => i !== index);
      const newIndex = newFiles.length === 0 ? -1 : Math.min(s.activeIndex, newFiles.length - 1);
      const activeEntry = newIndex >= 0 ? newFiles[newIndex] : null;
      return {
        openFiles: newFiles,
        activeIndex: newIndex,
        activeFilePath: activeEntry?.filePath ?? null,
        schema: activeEntry?.schema ?? null,
        selectedMessageType: activeEntry?.schema.messages[0]?.full_name ?? null,
      };
    }),
  setActiveIndex: (index) =>
    set((s) => {
      const entry = s.openFiles[index];
      return {
        activeIndex: index,
        activeFilePath: entry?.filePath ?? null,
        schema: entry?.schema ?? null,
        selectedMessageType: entry?.schema.messages[0]?.full_name ?? null,
        hexPreview: "",
        encodeError: null,
      };
    }),
  // ... remaining actions
  reset: () => set({ ...INITIAL_STATE }),
}));
```
