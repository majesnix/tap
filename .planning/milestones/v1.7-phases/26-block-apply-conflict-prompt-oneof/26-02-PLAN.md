```typescript
export type ApplyItemKind = "scalar" | "enum" | "well_known" | "map" | "oneof";

export type ConflictItemKind =
  | 'map_key_collision'
  | 'oneof_dirty_subfield'
  | 'oneof_branch_switch';

export type ConflictItem = {
  fieldName: string;
  blockValue: unknown;
  currentValue: unknown;
  kind: ConflictItemKind;
  subFieldName?: string;           // oneof_dirty_subfield only
  currentBranch?: string;          // oneof_branch_switch only
  blockBranch?: string;            // oneof_branch_switch only
  collisionKey?: string;           // map_key_collision only
  nonCollidingBlockRows?: unknown[];  // map_key_collision only; all block rows for this field that did NOT collide
  fieldLabel?: string;             // UI display label (field.label ?? fieldName)
  subFieldLabel?: string;          // UI display label for dirty-subfield rows
};

export type ConflictChoices = Record<string, 'skip' | 'overwrite'>;
// Compound key format:
//   map_key_collision:      "{fieldName}:{collisionKey}"
//   oneof_dirty_subfield:   "{fieldName}:{subFieldName}"
//   oneof_branch_switch:    "{fieldName}"  (bare field name)

export type ApplyBlockRef = {
  buildPlan: (blockValues: Record<string, unknown>) => ApplyPlan;
  commitApply: (plan: ApplyPlan, choices?: ConflictChoices) => void;
};

export type ApplyPlan = {
  toApply: ApplyItem[];
  conflicts: ConflictItem[];
  unknownKeys: string[];
};
```

IMPORTANT invariant from Plan 26-01: when any block row collides with an existing map key, buildApplyPlan emits ZERO toApply items for that map field. All data flows through conflicts (ConflictItems with collisionKey). The non-colliding block rows are carried in nonCollidingBlockRows on each ConflictItem for that field. commitApply Phase B must perform one atomic merge per map field that has overwrite choices.
