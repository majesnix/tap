```typescript
// lines 103-118: optimistic-write + rollback pattern
async renamePlan(id: string, name: string): Promise<void> {
  const prev = get().plans;
  set({ plans: prev.map(p => p.id === id ? { ...p, name } : p) });
  try {
    await persistPlans(get().plans);
  } catch (err) {
    set({ plans: prev });
    throw err;
  }
},
```
