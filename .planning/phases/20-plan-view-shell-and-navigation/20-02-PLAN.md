```tsx
export default function App() {
  const [viewMode, setViewMode] = useState<"main" | "plans">("main");

    void usePlanStore.getState().loadPlans();
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

    <ThemeProvider attribute="class" defaultTheme="system" enableSystem>

      <UpdateChecker />

        : <PlanView onViewChange={setViewMode} />
      }
      <Toaster />
    </ThemeProvider>
  );
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-20-02-01 | Spoofing | Plans nav button | accept | viewMode is local client state. No authentication boundary. This is a dev tool with no multi-user session concept. |
| T-20-02-02 | Information Disclosure | loadPlans() at App mount | accept | Plans data is developer-authored local data in plans.json on the developer's own machine. No credentials or PII. |
| T-20-02-03 | Denial of Service | loadPlans() error on mount | mitigate | usePlanStore.loadPlans() uses optimistic rollback and catches errors internally. A load failure leaves `plansLoaded: false` and `plans: []` — the UI renders an empty list, not a crash. No additional handling needed at App level. |
