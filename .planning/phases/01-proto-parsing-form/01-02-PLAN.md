```typescript
import { Controller, useFormContext } from "react-hook-form";
import { z } from "zod";
import { Input } from "@/components/ui/input";
import { Checkbox } from "@/components/ui/checkbox";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import type { FieldSchema, ScalarKind } from "@/lib/types";

interface ScalarFieldProps {
  field: FieldSchema;
  fieldPath: string;
}

// Zod schemas per scalar kind (RESEARCH.md Form Validation Strategy)
function getZodSchema(scalar: ScalarKind) {
  switch (scalar) {
    case "int32":
    case "sint32":
    case "sfixed32":
      return z.number().int().min(-2147483648).max(2147483647);
    case "uint32":
    case "fixed32":
      return z.number().int().min(0).max(4294967295);
    case "int64":
    case "sint64":
    case "sfixed64":
      return z.string().regex(/^-?\d+$/, "Must be an integer");
    case "uint64":
    case "fixed64":
      return z.string().regex(/^\d+$/, "Must be a non-negative integer");
    case "float":
    case "double":
      return z.number();
    case "bool":
      return z.boolean();
    case "string":
    case "bytes":
    default:
      return z.string();
  }
}

// Input type per scalar kind
function getInputType(scalar: ScalarKind): "text" | "number" | "checkbox" {
  if (scalar === "bool") return "checkbox";
  if (["int64", "uint64", "sint64", "fixed64", "sfixed64", "string", "bytes"].includes(scalar)) return "text";
  return "number";
}

export function ScalarField({ field, fieldPath }: ScalarFieldProps) {
  const { control } = useFormContext();
  const scalar = (field.kind as { type: "scalar"; scalar: ScalarKind }).scalar;
  const inputType = getInputType(scalar);
  const zodSchema = getZodSchema(scalar);

  const validate = (value: unknown) => {
    const result = zodSchema.safeParse(value);
    if (!result.success) {
      return result.error.issues[0]?.message ?? "Invalid value";
    }
    return true;
  };

  return (
    <div className="flex flex-col gap-1 mb-3">
      <div className="flex items-center gap-2">
        <Label className="text-xs font-semibold text-foreground" htmlFor={fieldPath}>
          {field.label}
        </Label>
        <Badge variant="outline" className="text-xs px-1.5 py-0">
          {scalar}
        </Badge>
        {scalar === "bytes" && (
          <Badge variant="secondary" className="text-xs px-1.5 py-0">
            bytes (base64)
          </Badge>
        )}
      </div>

      <Controller
        name={fieldPath}
        control={control}
        defaultValue={field.default_value ?? (inputType === "checkbox" ? false : inputType === "number" ? 0 : "")}
        rules={{ validate }}
        render={({ field: rhfField, fieldState }) => {
          if (inputType === "checkbox") {
            return (
              <Checkbox
                id={fieldPath}
                checked={!!rhfField.value}
                onCheckedChange={rhfField.onChange}
              />
            );
          }
          return (
            <Input
              id={fieldPath}
              type={inputType}
              value={rhfField.value ?? ""}
              onChange={(e) => {
                const raw = e.target.value;
                // For number inputs, parse to number; for text, keep as string
                if (inputType === "number") {
                  rhfField.onChange(raw === "" ? "" : Number(raw));
                } else {
                  rhfField.onChange(raw);
                }
              }}
              onBlur={rhfField.onBlur}
              aria-invalid={!!fieldState.error}
              className={fieldState.error ? "border-destructive" : ""}
            />
          );
        }}
      />

      {/* Inline validation error — FORM-06 */}
      <Controller
        name={fieldPath}
        control={control}
        render={({ fieldState }) =>
          fieldState.error ? (
            <p className="text-xs text-destructive" role="alert">
              {field.label}: {fieldState.error.message}
            </p>
          ) : null
        }
      />
    </div>
  );
}
```
