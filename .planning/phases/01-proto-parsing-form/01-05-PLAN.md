**Step 2 — Implement src/components/form/WellKnownTypeField.tsx (replace stub body, keep interface):**
```typescript
import React from "react";
import { Controller, useFormContext } from "react-hook-form";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import type { FieldSchema } from "@/lib/types";

export interface WellKnownTypeFieldProps {
  field: FieldSchema;
  fieldPath: string;
}

export function WellKnownTypeField({ field, fieldPath }: WellKnownTypeFieldProps) {
  const { control } = useFormContext();
  const wkt = (field.kind as { type: "well_known"; wkt: string }).wkt;

  // Duration validation regex (RESEARCH.md FORM-09)
  const DURATION_PATTERN = /^(\d+h)?(\d+m)?(\d+(\.\d+)?s)?$/;

  return (
    <div className="flex flex-col gap-1 mb-3">
      <div className="flex items-center gap-2">
        <Label className="text-xs font-semibold" htmlFor={fieldPath}>
          {field.label}
        </Label>
        <Badge variant="outline" className="text-xs px-1.5 py-0">wkt</Badge>
        {wkt !== "Timestamp" && wkt !== "Duration" && (
          <Badge variant="secondary" className="text-xs px-1.5 py-0 max-w-[200px] truncate">
            {wkt}
          </Badge>
        )}
      </div>

      {wkt === "Timestamp" && (
        <Controller
          name={fieldPath}
          control={control}
          defaultValue=""
          render={({ field: rhfField }) => (
            <input
              id={fieldPath}
              type="datetime-local"
              value={rhfField.value}
              onChange={rhfField.onChange}
              onBlur={rhfField.onBlur}
              className="flex h-9 w-full rounded-md border border-input bg-transparent px-3 py-1 text-sm shadow-sm transition-colors placeholder:text-muted-foreground focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-ring disabled:cursor-not-allowed disabled:opacity-50"
            />
          )}
        />
      )}

      {wkt === "Duration" && (
        <Controller
          name={fieldPath}
          control={control}
          defaultValue=""
          rules={{
            validate: (val: string) => {
              if (!val) return true; // empty is valid (zero duration)
              return DURATION_PATTERN.test(val) || "Must be a duration like 1h30m, 90s, or 2h15m30s";
            },
          }}
          render={({ field: rhfField, fieldState }) => (
            <>
              <Input
                id={fieldPath}
                type="text"
                placeholder="e.g. 1h30m"
                value={rhfField.value}
                onChange={rhfField.onChange}
                onBlur={rhfField.onBlur}
                aria-invalid={!!fieldState.error}
                className={fieldState.error ? "border-destructive" : ""}
              />
              {fieldState.error && (
                <p className="text-xs text-destructive" role="alert">
                  {field.label}: {fieldState.error.message}
                </p>
              )}
            </>
          )}
        />
      )}

      {wkt !== "Timestamp" && wkt !== "Duration" && (
        // Fallback for all other WKTs including google.protobuf.Any (G-8: no special Any handling)
        <Controller
          name={fieldPath}
          control={control}
          defaultValue=""
          render={({ field: rhfField }) => (
            <Input
              id={fieldPath}
              type="text"
              value={rhfField.value}
              onChange={rhfField.onChange}
              onBlur={rhfField.onBlur}
              placeholder={`${wkt} value`}
            />
          )}
        />
      )}
    </div>
  );
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| datetime-local input value → Rust encode | ISO 8601 string crosses IPC; Rust must parse it to seconds+nanos |
| Duration string → Rust encode | Duration string crosses IPC; Rust parses to seconds+nanos |
| tauri-plugin-store → filesystem | Include paths written to app data directory |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | WellKnownTypeField Timestamp — ISO 8601 string from browser input | mitigate | `datetime-local` browser input constrains format to YYYY-MM-DDTHH:MM (browser enforces). Rust receives this string and parses with standard ISO 8601 parser. Invalid strings (if somehow submitted) cause `AppError::EncodeError` returned to frontend. Two-layer: browser input type + Rust parse. |
| T-05-02 | Tampering | WellKnownTypeField Duration — user-entered duration string | mitigate | Frontend regex `^(\d+h)?(\d+m)?(\d+(\.\d+)?s)?$` validates before encode call. Rust additionally validates the parsed duration for range. Invalid input returns `AppError::EncodeError` displayed as inline error. |
| T-05-03 | Information Disclosure | tauri-plugin-store writes include paths to filesystem | accept | Include paths are filesystem paths the user explicitly entered. They are not secrets. The store writes to the app data directory (macOS: `~/Library/Application Support`, Windows: `%APPDATA%`). No sensitive data is stored. Risk accepted. |
| T-05-04 | Tampering | Include path dialog — user can inject arbitrary filesystem paths as include dirs | accept | Include paths are passed to `protox::Compiler::new(&include_paths)` as search roots. protox only reads `.proto` files from these dirs. A malicious include path causes protox to read unexpected .proto files — output is a schema JSON, not code execution. The user controls their own include paths. Risk accepted. |
</threat_model>
