**src/components/response/ResponseHexSection.tsx — complete replacement:**

Remove the `useResponseStore` import and store-read entirely. Replace the component with the props-based version:

```tsx
import { toast } from "sonner";
import { Button } from "@/components/ui/button";
import { Copy } from "lucide-react";

interface ResponseHexSectionProps {
  hexString: string;
  decoded: Record<string, unknown> | null;
}

export function ResponseHexSection({ hexString, decoded }: ResponseHexSectionProps) {
  if (!hexString) return null;

  const handleCopyHex = async () => {
    try {
      await navigator.clipboard.writeText(hexString);
      toast("Hex copied", { duration: 2000 });
    } catch {
      toast.error("Copy failed — clipboard access denied", { duration: 2000 });
    }
  };

  const handleCopyJson = async () => {
    if (!decoded) return;
    try {
      await navigator.clipboard.writeText(JSON.stringify(decoded, null, 2));
      toast("JSON copied", { duration: 2000 });
    } catch {
      toast.error("Copy failed — clipboard access denied", { duration: 2000 });
    }
  };

  return (
    <div className="p-4 border-t border-border space-y-2">
      <div className="flex items-center justify-between">
        <span className="text-xs font-semibold text-muted-foreground uppercase tracking-wide">
          Raw Hex
        </span>
        <Button variant="ghost" size="sm" onClick={() => void handleCopyHex()}>
          <Copy className="w-3 h-3 mr-1" />
          Copy hex
        </Button>
      </div>
      <pre className="text-xs font-mono bg-muted rounded p-2 overflow-x-auto whitespace-pre-wrap break-all">
        {hexString}
      </pre>
      {decoded !== null && (
        <div className="flex justify-end">
          <Button variant="ghost" size="sm" onClick={() => void handleCopyJson()}>
            <Copy className="w-3 h-3 mr-1" />
            Copy JSON
          </Button>
        </div>
      )}
    </div>
  );
}
```

```tsx
import { describe, beforeEach, test, expect, vi } from "vitest";
import { render, screen, fireEvent, waitFor } from "@testing-library/react";

// Mock clipboard
const mockWriteText = vi.fn().mockResolvedValue(undefined);
Object.defineProperty(navigator, "clipboard", {
  value: { writeText: mockWriteText },
  writable: true,
});

// Mock sonner toast (vi.hoisted pattern — required for Vitest hoisting)
const { mockToast } = vi.hoisted(() => ({ mockToast: vi.fn() }));
vi.mock("sonner", () => ({ toast: Object.assign(mockToast, { error: vi.fn() }) }));

beforeEach(() => {
  vi.clearAllMocks();
});

describe("ResponseHexSection", () => {
  test("renders nothing when hexString is empty", () => {
    const { container } = render(
      <ResponseHexSection hexString="" decoded={null} />
    );
    expect(container.firstChild).toBeNull();
  });

  test("renders hex display without JSON copy button when decoded is null", () => {
    render(<ResponseHexSection hexString="0a 05 62 6f 62" decoded={null} />);
    expect(screen.getByText("0a 05 62 6f 62")).toBeInTheDocument();
    expect(screen.getByText("Copy hex")).toBeInTheDocument();
    expect(screen.queryByText("Copy JSON")).toBeNull();
  });

  test("renders JSON copy button when decoded is provided", () => {
    render(
      <ResponseHexSection hexString="0a 05" decoded={{ name: "test" }} />
    );
    expect(screen.getByText("Copy hex")).toBeInTheDocument();
    expect(screen.getByText("Copy JSON")).toBeInTheDocument();
  });

  test("copies hex string to clipboard on Copy hex click", async () => {
    render(<ResponseHexSection hexString="0a 05" decoded={null} />);
    fireEvent.click(screen.getByText("Copy hex"));
    await waitFor(() => {
      expect(mockWriteText).toHaveBeenCalledWith("0a 05");

    fireEvent.click(screen.getByText("Copy JSON"));
    await waitFor(() => {
      expect(mockWriteText).toHaveBeenCalledWith(
        JSON.stringify(decoded, null, 2)
      );
    });
  });
});
```
