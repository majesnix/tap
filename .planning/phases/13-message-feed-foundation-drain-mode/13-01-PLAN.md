```rust
/// A single message result from the drain loop.
/// D-14: Ack happens before decode for every message in the loop.
#[derive(serde::Serialize)]
#[serde(rename_all = "camelCase")]
pub struct DrainResult {
    pub routing_key: String,
    pub exchange: String,
    pub content_type: Option<String>,
    pub timestamp: Option<u64>,
    pub decoded: Option<serde_json::Value>,
    pub hex_string: String,
    pub error: Option<String>,
}

/// Outcome of a drain_messages call.
/// partial_error is set when an AMQP error occurs mid-loop after at least one
/// message was already acked — already-collected messages are returned rather than discarded.
#[derive(serde::Serialize)]
#[serde(rename_all = "camelCase")]
pub struct DrainOutcome {
    pub messages: Vec<DrainResult>,
    pub partial_error: Option<String>,
}

/// drain_messages: Drain up to `count` messages from a queue in one AMQP connection.
///
/// SECURITY: AMQP URI built in tight scope; password dropped before connecting.
/// D-13/D-14: Ack-before-decode per message; partial_error on mid-loop AMQP failure.
/// D-15: consume_message is retained separately for compatibility.
#[tauri::command]
pub async fn drain_messages(
    app: tauri::AppHandle,
    profile_name: String,
    queue_name: String,
    message_type_name: String,
    count: u32,
    pool_state: tauri::State<'_, std::sync::Mutex<Option<prost_reflect::DescriptorPool>>>,
) -> Result<DrainOutcome, crate::error::AppError> {
    // Validate count: must be 1–500 (FIFO cap is 500 per D-17)
    if count == 0 || count > 500 {
        return Err(crate::error::AppError::AmqpError(
            "drain count must be between 1 and 500".to_string(),
        ));
    }

    // Step 1: Clone pool FIRST (before any .await — MutexGuard is not Send)
    let pool = {
        let guard = pool_state
            .lock()
            .map_err(|_| crate::error::AppError::EncodeError {
                field: "<root>".to_string(),
                message: "Internal state lock poisoned — restart the application".to_string(),
            })?;
        guard
            .as_ref()
            .ok_or_else(|| crate::error::AppError::EncodeError {
                field: "<root>".to_string(),
                message: "No proto file loaded".to_string(),
            })?
            .clone() // O(1) — Arc-backed
    }; // guard drops here

    // Step 2: Load credentials (sync, no await)
    let (profile, password) =
        crate::commands::connection::load_profile_with_password(&app, &profile_name)?;

    // Step 3: Connect in tight URI scope (SECURITY: password and URI dropped before result inspection)
    let conn = {
        let uri = crate::profiles::build_amqp_uri(
            &profile.host,
            profile.port,
            &profile.vhost,
            &profile.username,
            &password,
        );
        // password is no longer needed — drop it before connecting
        drop(password);
        let result = tokio::time::timeout(
            std::time::Duration::from_secs(10),
            lapin::Connection::connect(&uri, lapin::ConnectionProperties::default()),
        )
        .await;
        // uri drops here, before result inspection
        result
            .map_err(|_| {
                crate::error::AppError::AmqpError(
                    "Drain connection timed out (10s)".to_string(),
                )
            })?
            .map_err(|_| {
                crate::error::AppError::AmqpError(
                    "AMQP connection failed — check host, port, vhost, and credentials"
                        .to_string(),
                )
            })?
    };

    // Step 4: Create channel (close conn on error)
    let channel = match conn.create_channel().await {
        Ok(ch) => ch,
        Err(e) => {
            let _ = conn.close(0, "".into()).await;
            return Err(crate::error::AppError::AmqpError(e.to_string()));
        }
    };

    // Step 5: Drain loop — up to `count` messages, stop early on empty queue or AMQP error
    let mut messages: Vec<DrainResult> = Vec::new();
    let mut partial_error: Option<String> = None;

    for _ in 0..count {
        // basic_get NON-BLOCKING — returns None when queue is empty
        let get_result = channel
            .basic_get(
                queue_name.as_str().into(),
                lapin::options::BasicGetOptions::default(),
            )
            .await;

        let msg = match get_result {
            Err(e) => {
                // Mid-loop AMQP error: set partial_error, stop loop.
                // Already-acked messages in `messages` are preserved and returned.
                partial_error = Some(format!("AMQP error after {} message(s): {}", messages.len(), e));
                break;
            }
            Ok(None) => {
                // Queue empty — stop loop, return what we have
                break;
            }
            Ok(Some(msg)) => msg,
        };

        // Extract metadata BEFORE acking
        let delivery_tag = msg.delivery_tag;
        let payload: Vec<u8> = msg.delivery.data.clone();
        let hex_string = bytes_to_hex(&payload);

        // CRITICAL: ShortString → String (Pitfall 7)
        let routing_key = msg.delivery.routing_key.to_string();
        let exchange = msg.delivery.exchange.to_string();
        let content_type = msg.delivery.properties.content_type().as_ref().map(|s| s.to_string());
        // *deref Option<u64> — Pitfall 3: timestamp is optional; None is valid
        let timestamp: Option<u64> = *msg.delivery.properties.timestamp();

        // ACK BEFORE DECODE — D-14: always ack, even if decode will fail
        if let Err(e) = channel
            .basic_ack(delivery_tag, lapin::options::BasicAckOptions::default())
            .await
        {
            // Ack failure is a hard error — stop the loop
            partial_error = Some(format!(
                "Ack failed after {} message(s): {}",
                messages.len(),
                e
            ));
            break;
        }

        // Decode (synchronous — pool already cloned, no awaits needed here)
        let (decoded, decode_error) = match pool.get_message_by_name(&message_type_name) {
            None => (
                None,
                Some(format!(
                    "Message type '{}' not found in loaded schema",
                    message_type_name
                )),
            ),
            Some(msg_desc) => {
                match prost_reflect::DynamicMessage::decode(msg_desc, payload.as_ref()) {
                    Ok(dyn_msg) => {
                        let mut buf = Vec::new();
                        let mut ser = serde_json::Serializer::new(&mut buf);
                        let opts = prost_reflect::SerializeOptions::new()
                            .use_proto_field_name(true)
                            .stringify_64_bit_integers(true);
                        match dyn_msg.serialize_with_options(&mut ser, &opts) {
                            Ok(()) => match serde_json::from_slice::<serde_json::Value>(&buf) {
                                Ok(v) => (Some(v), None),
                                Err(e) => (None, Some(format!("Decode failed: {}. Showing raw bytes.", e))),
                            },
                            Err(e) => (None, Some(format!("Decode failed: {}. Showing raw bytes.", e))),
                        }
                    }
                    Err(e) => (
                        None,
                        Some(format!("Decode failed: {}. Showing raw bytes.", e)),
                    ),
                }
            }
        };

        messages.push(DrainResult {
            routing_key,
            exchange,
            content_type,
            timestamp,
            decoded,
            hex_string,
            error: decode_error,
        });
    }

    // Step 6: Close connection after loop (always)
    let _ = conn.close(0, "".into()).await;

    Ok(DrainOutcome {
        messages,
        partial_error,
    })
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Frontend → drain_messages command | count parameter crosses the IPC boundary; must be validated in Rust |
| AMQP server → DrainResult | routing_key, exchange, content_type are arbitrary broker-controlled strings |
| Rust → Frontend | DrainResult fields serialized via serde; no HTML rendering in this plan |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-01 | Tampering | drain_messages count parameter | mitigate | Validate 1 ≤ count ≤ 500 in Rust before opening AMQP connection; return AppError::AmqpError if out of range |
| T-13-02 | Information Disclosure | AMQP URI (cleartext password) | mitigate | URI built in tight scope; password dropped with explicit `drop(password)` before `Connection::connect`; URI variable dropped before result inspection |
| T-13-03 | Denial of Service | Unbound drain loop | mitigate | count validated ≤ 500; each basic_get has per-call timeout inherited from lapin connection (10s connect timeout) |
| T-13-04 | Information Disclosure | Error messages leaking URI/credentials | mitigate | All AMQP errors returned as generic strings ("check host, port, vhost, and credentials") — no URI interpolation in error messages |
| T-13-05 | Repudiation | Acked messages with failed decode | accept | Ack-before-decode is a deliberate design decision (D-14); decode errors are surfaced inline in DrainResult.error — no silent data loss |
</threat_model>
