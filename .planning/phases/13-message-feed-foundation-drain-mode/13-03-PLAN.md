**Step A — Rewrite src/stores/useResponseStore.ts:**

```typescript
import { create } from "zustand";
import type { FeedMessage, DrainResult } from "@/lib/types";

export const FEED_MAX_SIZE = 500;

interface ResponseStore {
  queueList: string[];
  isLiveMode: boolean;
  selectedQueue: string;
  isLoading: boolean;
  messages: FeedMessage[];
  lastReadAt: number | null;
  queueDepth: number | null;

  setQueueList: (queues: string[], isLive: boolean) => void;
  setSelectedQueue: (queue: string) => void;
  setIsLoading: (loading: boolean) => void;
  appendMessages: (incoming: DrainResult[]) => void;
  clearMessages: () => void;
  setLastReadAt: (ts: number | null) => void;
  setQueueDepth: (depth: number | null) => void;
  reset: () => void;
}

const INITIAL_STATE = {
  queueList: [] as string[],
  isLiveMode: false,
  selectedQueue: "",
  isLoading: false,
  messages: [] as FeedMessage[],
  lastReadAt: null as number | null,
  queueDepth: null as number | null,
} as const;

export const useResponseStore = create<ResponseStore>((set) => ({
  ...INITIAL_STATE,
  setQueueList: (queueList, isLiveMode) => set({ queueList, isLiveMode }),
  setSelectedQueue: (selectedQueue) => set({ selectedQueue }),
  setIsLoading: (isLoading) => set({ isLoading }),
  appendMessages: (incoming: DrainResult[]) =>
    set((state) => {
      const asFeedMessages: FeedMessage[] = incoming.map((r) => ({
        id: crypto.randomUUID(),
        routingKey: r.routingKey,
        exchange: r.exchange,
        contentType: r.contentType,
        timestamp: r.timestamp,
        decoded: r.decoded,
        hexString: r.hexString,
        error: r.error,
      }));
      const combined = [...asFeedMessages, ...state.messages];
      return { messages: combined.slice(0, FEED_MAX_SIZE) };
    }),
  clearMessages: () => set({ messages: [] }),
  setLastReadAt: (lastReadAt) => set({ lastReadAt }),
  setQueueDepth: (queueDepth) => set({ queueDepth }),
  reset: () => set({ ...INITIAL_STATE }),
}));
```

Replace the entire file. Key changes: `onRead: () => void` → `onDrain: (count: number) => void`; add `drainCount` local state; add drain count `<input>`; rename Read → Drain throughout:

```tsx
import { useState, useEffect } from "react";
import { Loader2 } from "lucide-react";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip";
import { useConnectionStore } from "@/stores/useConnectionStore";
import { useResponseStore } from "@/stores/useResponseStore";
import { fetchQueues, fetchQueueDepth } from "@/lib/ipc";

interface ResponseQueuePickerProps {
  onDrain: (count: number) => void;
}

export function ResponseQueuePicker({ onDrain }: ResponseQueuePickerProps) {
  const [managementAuthError, setManagementAuthError] = useState<string | null>(null);
  const [drainCount, setDrainCount] = useState<number>(10);

  const { activeProfileName, connectionStatus } = useConnectionStore();
  const {
    queueList,
    isLiveMode,
    selectedQueue,
    isLoading,
    lastReadAt,
    queueDepth,
    setQueueList,
    setSelectedQueue,
    setQueueDepth,
  } = useResponseStore();

  // Queue fetch on tab focus — useEffect with [activeProfileName] dep
  useEffect(() => {
    if (!activeProfileName) return;
    let cancelled = false;

    const fetch = async () => {
      try {
        const qs = await fetchQueues(activeProfileName);
        if (cancelled) return;
        setManagementAuthError(null);
        setQueueList(qs, true);
      } catch (err: unknown) {
        if (cancelled) return;
        const errMsg = err instanceof Error ? err.message : String(err);
        if (errMsg.startsWith("Management API authentication failed")) {
          setManagementAuthError(errMsg);
          setQueueList([], false);
        } else {
          setManagementAuthError(null);
          setQueueList([], false);
        }
      }
    };
    void fetch();
    return () => {
      cancelled = true;
    };
  }, [activeProfileName, setQueueList, setManagementAuthError]);

  // Refresh depth whenever the selected queue or last read changes
  useEffect(() => {
    if (!activeProfileName || !selectedQueue) {
      setQueueDepth(null);
      return;
    }
    let cancelled = false;
    void (async () => {
      try {
        const depth = await fetchQueueDepth(activeProfileName, selectedQueue);
        if (!cancelled) setQueueDepth(depth);
      } catch {
        if (!cancelled) setQueueDepth(null);
      }
    })();
    return () => {
      cancelled = true;
    };
  }, [activeProfileName, selectedQueue, lastReadAt, setQueueDepth]);

  const handleDrainClick = () => {
    // Clamp and default before invoking
    const safeCount = Number.isNaN(drainCount) || drainCount < 1
      ? 10
      : Math.min(drainCount, 500);
    onDrain(safeCount);
  };

  return (
    <div className="px-4 py-2 border-b border-border flex items-center gap-2 flex-wrap">
      {/* Live dropdown vs Manual text input */}
      {isLiveMode ? (
        <Select value={selectedQueue} onValueChange={setSelectedQueue}>
          <SelectTrigger className="w-48">
            <SelectValue placeholder="Select queue…" />
          </SelectTrigger>
          <SelectContent position="popper" className="max-h-60">
            {queueList.map((name) => (
              <SelectItem key={name} value={name}>
                {name}
              </SelectItem>
            ))}
          </SelectContent>
        </Select>
      ) : (
        <Input
          placeholder="Queue name"
          className="w-48"
          value={selectedQueue}
          onChange={(e) => setSelectedQueue(e.target.value)}
        />
      )}

      {/* Live / Manual / Auth-error badge */}
      {managementAuthError ? (
        <Badge variant="destructive" className="text-xs">
          {managementAuthError}
        </Badge>
      ) : isLiveMode ? (
        <Badge variant="outline" className="text-xs gap-1">
          <span className="w-2 h-2 rounded-full bg-emerald-500" />
          Live
        </Badge>
      ) : (
        <Badge variant="outline" className="text-xs gap-1">
          <span className="w-2 h-2 rounded-full bg-amber-500" />
          Manual
        </Badge>
      )}

      {/* Queue depth pill */}
      {queueDepth !== null && (
        <Badge variant="secondary" className="text-xs tabular-nums">
          {queueDepth === 0 ? "empty" : `${queueDepth} msg${queueDepth === 1 ? "" : "s"}`}
        </Badge>
      )}

      {/* Drain count input — UI-SPEC: w-12 h-9 text-sm text-center */}
      <input
        type="number"
        aria-label="Drain count"
        min={1}
        max={500}
        value={drainCount}
        onChange={(e) => setDrainCount(parseInt(e.target.value, 10))}
        onBlur={() => {
          if (Number.isNaN(drainCount) || drainCount < 1) setDrainCount(1);
          else if (drainCount > 500) setDrainCount(500);
        }}
        className="w-12 h-9 text-sm text-center rounded-md border border-input bg-background px-1"
      />

      {/* Drain button — disabled+tooltip when disconnected */}
      {connectionStatus === "connected" ? (
        <Button
          variant="default"
          disabled={!selectedQueue.trim() || isLoading}
          onClick={handleDrainClick}
        >
          {isLoading ? <Loader2 className="w-4 h-4 mr-2 animate-spin" /> : null}
          Drain
        </Button>
      ) : (
        <TooltipProvider>
          <Tooltip>
            <TooltipTrigger asChild>
              <span>
                <Button variant="default" disabled>
                  Drain
                </Button>
              </span>
            </TooltipTrigger>
            <TooltipContent>Connect to a RabbitMQ profile to drain messages.</TooltipContent>
          </Tooltip>
        </TooltipProvider>
      )}
    </div>
  );
}
```

```tsx
function formatTimestamp(timestamp: number | null): string {
  if (timestamp === null) return "—"; // em dash
  return new Date(timestamp * 1000).toLocaleTimeString([], {
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false,
  });
}

  const timestamp = formatTimestamp(message.timestamp);

        {message.routingKey} &bull; {message.exchange} &bull; {contentType} &bull; {timestamp}

        <ResponseHexSection hexString={message.hexString} decoded={message.decoded} />
```

**Step B — Create src/components/response/MessageFeedTab.tsx:**

```tsx
import { Trash2 } from "lucide-react";
import { toast } from "sonner";
import { Accordion } from "@/components/ui/accordion";
import { Button } from "@/components/ui/button";
import { ScrollArea } from "@/components/ui/scroll-area";
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip";
import { useConnectionStore } from "@/stores/useConnectionStore";
import { useProtoStore } from "@/stores/useProtoStore";
import { useResponseStore } from "@/stores/useResponseStore";
import { drainMessages } from "@/lib/ipc";
import { ResponseQueuePicker } from "./ResponseQueuePicker";
import { MessageFeedRow } from "./MessageFeedRow";

export function MessageFeedTab() {
  const { activeProfileName } = useConnectionStore();
  const { selectedMessageType } = useProtoStore();
  const {
    selectedQueue,
    messages,
    setIsLoading,
    appendMessages,
    clearMessages,
    setLastReadAt,
  } = useResponseStore();

  const handleDrain = async (count: number) => {
    if (!activeProfileName || !selectedMessageType || !selectedQueue.trim()) return;
    setIsLoading(true);
    try {
      const outcome = await drainMessages(
        activeProfileName,
        selectedQueue,
        selectedMessageType,
        count,
      );

      if (outcome.messages.length === 0) {
        // D-03/D-12: empty drain — toast only, feed unchanged
        toast.info("Queue is empty");
      } else {
        appendMessages(outcome.messages);
        // CONS-04: trigger queue depth refresh via lastReadAt side effect
        setLastReadAt(Date.now());
      }

      if (outcome.partialError !== null) {
        toast.error(`Drain stopped early: ${outcome.partialError}`);
      }
    } catch (err: unknown) {
      const msg = err instanceof Error ? err.message : String(err);
      toast.error(`Drain failed: ${msg}`);
    } finally {
      setIsLoading(false);
    }
  };

  const messageCount = messages.length;

  return (
    <div className="flex flex-col h-full">
      <ResponseQueuePicker onDrain={(count) => void handleDrain(count)} />

      {/* Feed header: message count left, Clear button right */}
      <div className="flex items-center justify-between px-4 py-2 bg-muted">
        <span className="text-sm font-semibold">
          {messageCount === 0 ? "No messages" : `${messageCount} messages`}
        </span>
        <TooltipProvider>
          <Tooltip>
            <TooltipTrigger asChild>
              <Button
                variant="ghost"
                size="icon"
                onClick={clearMessages}
                aria-label="Clear feed"
              >
                <Trash2 className="w-4 h-4" />
              </Button>
            </TooltipTrigger>
            <TooltipContent>Clear feed</TooltipContent>
          </Tooltip>
        </TooltipProvider>
      </div>

      <ScrollArea className="flex-1 overflow-hidden">
        {messageCount === 0 ? (
          <p className="text-xs text-muted-foreground p-4">
            Select a queue and click Drain
          </p>
        ) : (
          <Accordion type="single" collapsible className="w-full">
            {messages.map((msg) => (
              <MessageFeedRow key={msg.id} message={msg} />
            ))}
          </Accordion>
        )}
      </ScrollArea>
    </div>
  );
}
```
