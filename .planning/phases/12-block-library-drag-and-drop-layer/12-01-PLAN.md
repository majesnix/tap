    3. Add a new useEffect AFTER the existing resetRef useEffect:
       ```typescript
       useEffect(() => {
         if (applyBlockRef) {
           applyBlockRef.current = (blockValues: Record<string, unknown>): string[] => {
             const skipped: string[] = [];
             const eligibleFields = new Set(
               message.fields
                 .filter(f => !f.repeated && (f.kind.type === 'scalar' || f.kind.type === 'enum'))
                 .map(f => f.name)
             );
             for (const [key, value] of Object.entries(blockValues)) {
               if (!eligibleFields.has(key)) {
                 skipped.push(key);
               } else if (methods.formState.dirtyFields[key]) {
                 // field is dirty — user edited it; do not overwrite (BLK-07)
                 // NOTE: uses dirtyFields per D-03. This means a field where user
                 // typed a value back to its default is NOT protected (value-based
                 // semantics). If interaction-history semantics are wanted, replace
                 // formState.dirtyFields with formState.touchedFields.
               } else {
                 methods.setValue(key, value as Parameters<typeof methods.setValue>[1]);
                 // Default shouldDirty: false — block-filled fields are NOT marked dirty,
                 // so a second block drop can overwrite them (intentional per D-03).
               }
             }
             return skipped;
           };
         }
         return () => {
           if (applyBlockRef) applyBlockRef.current = null;
         };
       }, [applyBlockRef, methods, message]);
       ```
    
    Dependency array must include `message` because `message.fields` drives the eligible set.

    Do NOT pass `{ shouldDirty: true }` to setValue — the default shouldDirty: false is correct.
