    ```tsx
    import { useState, useEffect } from "react";
    import { useTheme } from "next-themes";
    import { Sun, Moon, Monitor } from "lucide-react";
    import { Button } from "@/components/ui/button";

    type ThemeMode = "system" | "light" | "dark";

    const CYCLE_ORDER: ThemeMode[] = ["system", "light", "dark"];

    const ICONS: Record<ThemeMode, React.ReactNode> = {
      system: <Monitor className="size-4" />,
      light: <Sun className="size-4" />,
      dark: <Moon className="size-4" />,
    };

    const LABELS: Record<ThemeMode, string> = {
      system: "System theme",
      light: "Light theme",
      dark: "Dark theme",
    };

    export function ThemeToggle() {
      const [mounted, setMounted] = useState(false);
      const { theme, setTheme } = useTheme();

      useEffect(() => setMounted(true), []);

      // Return same-size placeholder before mount to avoid layout shift (Pitfall 1)
      if (!mounted) {
        return <Button variant="ghost" size="icon" className="size-8" disabled />;
      }

      const current = (theme as ThemeMode) ?? "system";
      const nextIndex = (CYCLE_ORDER.indexOf(current) + 1) % CYCLE_ORDER.length;
      const nextMode = CYCLE_ORDER[nextIndex];

      return (
        <Button
          variant="ghost"
          size="icon"
          className="size-8"
          onClick={() => setTheme(nextMode)}
          aria-label={LABELS[current]}
          title={LABELS[current]}
        >
          {ICONS[current]}
        </Button>
      );
    }
    ```

    Critical details (all load-bearing):
    - CYCLE_ORDER: ["system", "light", "dark"] — exact order per D-05
    - Icon map: Monitor for system, Sun for light, Moon for dark — per D-05
    - aria-label and title describe CURRENT mode (not next) — per UI-SPEC copywriting contract
    - mounted guard returns a DISABLED Button (same size-8 dimensions) — not null — to prevent layout shift
    - Icon className is "size-4" (16px) — per UI-SPEC spacing tokens
    - Button className is "size-8" (32px) — per UI-SPEC and button.tsx size="icon" default
  </action>
